// JWT и®ӨиҜҒиҝҮж»ӨеҷЁпјҲopencloud-authпјү
@Component
@RequiredArgsConstructor
public class JwtAuthenticationFilter extends OncePerRequestFilter {
    
    private final JwtUtil jwtUtil;
    private final RedisTemplate<String, String> redisTemplate;
    
    @Override
    protected void doFilterInternal(HttpServletRequest request, 
                                    HttpServletResponse response, 
                                    FilterChain chain) {
        // 1. д»Һ Header жҸҗеҸ– Token
        String token = extractToken(request);
        
        // 2. ж ЎйӘҢ Token жңүж•ҲжҖ§
        if (token != null && jwtUtil.validateToken(token)) {
            // 3. жЈҖжҹҘй»‘еҗҚеҚ•
            if (isTokenBlacklisted(token)) {
                throw new BusinessException("Token е·ІеӨұж•Ҳ");

            // 4. еҠ иҪҪз”ЁжҲ·жқғйҷҗ
            Authentication authentication = buildAuthentication(token);
            SecurityContextHolder.getContext().setAuthentication(authentication);
        }
        
        chain.doFilter(request, response);
    }
}

### 2.5 е…ЁеұҖй…ҚзҪ®дёҺзәҝзЁӢжұ  (opencloud-web)

`opencloud-web` дҪңдёәдё»е…ҘеҸЈпјҢз»ҹдёҖз®ЎзҗҶд»ҘдёӢе…ЁеұҖй…ҚзҪ®пјҡ

- **WebMvcConfig**пјҡй…ҚзҪ® CORS и·Ёеҹҹзӯ–з•ҘпјҲејҖеҸ‘зҺҜеўғ `*`пјҢз”ҹдә§зҺҜеўғжҢүйңҖй…ҚзҪ®пјүпјҢжіЁеҶҢ `TraceIdInterceptor` е®һзҺ°е…Ёй“ҫи·Ҝж—Ҙеҝ—иҝҪиёӘгҖӮ
- **RedisConfig**пјҡй…ҚзҪ® `RedisTemplate<String, Object>` дҪҝз”Ё `Jackson2JsonRedisSerializer`пјҢжӣҝд»Јй»ҳи®Өзҡ„ JDK еәҸеҲ—еҢ–пјҢдҫҝдәҺи·ЁиҜӯиЁҖе’ҢеҸҜи§ҶеҢ–жҹҘзңӢгҖӮ
- **MyBatisPlusConfig**пјҡжіЁеҶҢ `PaginationInnerInterceptor`пјҲPostgreSQL ж–№иЁҖпјүе’Ң `OptimisticLockerInnerInterceptor`пјҲд№җи§Ӯй”ҒпјүгҖӮ
- **AsyncConfig**пјҡй…ҚзҪ® `@Async` е…ЁеұҖејӮжӯҘзәҝзЁӢжұ пјҲж ёеҝғ10пјҢжңҖеӨ§50пјҢйҳҹеҲ—100пјүпјҢйҮҮз”Ё `CallerRunsPolicy` жӢ’з»қзӯ–з•ҘпјҢж”ҜжҢҒдјҳйӣ…еҒңжңәгҖӮ
- **SchedulerConfig**пјҡй…ҚзҪ® `@Scheduled` и°ғеәҰзәҝзЁӢжұ пјҲеӨ§е°Ҹ10пјүпјҢйҳІжӯўеӨҡдёӘе®ҡж—¶д»»еҠЎдә’зӣёйҳ»еЎһгҖӮ
- **SwaggerConfig**пјҡдҪҝз”Ё Knife4j е°Ҷ API жҢүжЁЎеқ—еҲҶз»„пјҲ01-и®ӨиҜҒжҺҲжқғгҖҒ02-зі»з»ҹз®ЎзҗҶгҖҒ03-дә‘е№іеҸ°з®ЎзҗҶгҖҒ04-зӣ‘жҺ§е‘ҠиӯҰпјүгҖӮ
